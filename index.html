Responsive font size

variables.scss
// layouts:	  main       s          m
$body-font-size:  18px,      20px,      22px;
$h1-size:         48px 1.2,  52px 1.1,  58px 1;
css output
body {
  font-size: 18px;
}

h1 {
  font-size: 48px;
  line-height: 1.2;
}

@media only screen and (min-width: 600px) {
  body {
    font-size: 20px;
  }
  h1 {
    font-size: 52px;
    line-height: 1.1;
  }
}

@media only screen and (min-width: 1000px) {
  body {
    font-size: 22px;
  }
  h1 {
    font-size: 58px;
    line-height: 1;
  }
}
